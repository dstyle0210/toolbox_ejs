# 프로젝트 폴더구조
## 루트폴더(```./```) 구조
```
toolbox-ejs/
├── .www            # 개발용 로컬서버를 위한 임시폴더
├── node_modules    # node 모듈
├── gulp_modules    # gulp 모듈
│   ├── servers     # gulp를 통한 서버들
│   ├── routes      # 서버용 라우터
│   ├── tasks       # gulp task 들(compile , watch 등)
│   └── utils       # gulp 모듈에 사용하는 공통유틸리티
├── src
│   ├── assets      # 컴파일이 필요한 리소스(scss , ts)
│   ├── mocks       # 템플릿용 데이터(json , xml 등)
│   ├── public      # 정적 리소스(이미지 , 벤더 등)
│   ├── templates   # UI 관련 템플릿 모듈 파일 (ejs , scss , ts)
│   └── pages       # 각 페이지별 EJS 파일 (ejs , scss , ts)
├── dist            
│   ├── css         # 제출용 CSS
│   ├── js          # 제출용 js
│   ├── venders     # 제출용 벤더
│   └── pages       # 제출용 html
├── gulpfile.ts
├── package.json
├── .gitignore
└── README.md
```
